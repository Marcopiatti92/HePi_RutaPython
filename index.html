EjercicioEscribir una aplicacion que reciba una cantidad infinita de numeros hasta decir basta, Luego que devuelva la suma de los numeros ingresados

                                                       
lista = []
print('ingrese nunmeros y para salir diga "basta"')
while True:
    valor= input ('ingrese Valor: ')
    if valor == 'basta':
        break
    else:
        try:
            valor = int(valor)
            lista.append(valor)
        except:
            print('El valor ingresado no es un int')
            exit()
resultado=0
for x in lista:
    resultado += x
print (resultado)
